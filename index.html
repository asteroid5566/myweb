            CONSOLE_FONT_INFOEX font = { sizeof(CONSOLE_FONT_INFOEX) };
            if (!GetCurrentConsoleFontEx(windowHandle, 0, &font)){
                return 0;
            }
            *size = font.dwFontSize;
            return 1;
        }
        int SetFontSize(HANDLE windowHandle, COORD size){
            CONSOLE_FONT_INFOEX font = { sizeof(CONSOLE_FONT_INFOEX) };
            if (!GetCurrentConsoleFontEx(windowHandle, 0, &font)){
                return 0;
            }
            font.dwFontSize = size;
            if (!SetCurrentConsoleFontEx(windowHandle, 0, &font)){
                return 0;
            }
            return 1;
        }
            
        //以下放入主函式
        HANDLE h = GetStdHandle(STD_OUTPUT_HANDLE);
        COORD size;
        if (GetFontSize(h, &size)){
            size.X += (SHORT)(size.X * .5);//Grow by 50%
            size.Y += (SHORT)(size.Y * .5);
            SetFontSize(h, size);
       }

  • %.pf               只會顯示小數點後P位數的值
  • #define 名稱 (1.0f/3.14159f)               定義marco(名稱通常全大寫)
  • %m.pX               會顯示M位數(少於的話會有空格，多餘的話會延長)，M前面加負號空格會跑到右邊
  • double asd= 10 / 3.;               隱性轉型
  • double asd= 10 / (double)3;               顯性轉型
  • &&且 ||或 !非               !(3>2)跟!3>2不一樣
  • 運算子順序: ! % + > != && || = 算數>關係>邏輯
  • %c 前面加空白的話就會讀第一個非空白的字元(跳過不可列印字元)
  • printf的修飾子EX: i=1234; printf(“%+08d”,i);               輸出: +0001234
  • char 字串[長度];               宣告字元陣列來儲存字串(最後一格是\0)
  • scanf(“%s”,字串);               注意不用加&運算子
  • fflush(stdin);               清除輸入緩衝區的資料
  • putchar(變數);               將字元輸出到螢幕
  • 變數=getchar();               打完要按Enter
  • 變數=getche();               讀取字元並回應在螢幕上 (下兩者需要conio.h，不須按Enter)
  • 變數=getch();               不顯示在螢幕
  • 判斷式成立會輸出1               EX: printf(“%d”,5>2);
  • a++               執行敘述後再加1
  • ++a               將a加一再執行敘述
